RawBill <-read_excel("hoa_don_cod_20191219120620_5dfb054c-dcd0-4c9e-af48-cc6e0a0a026e.xlsx", 
    sheet = "Hoa n COD")
    
# Danh sach COD quan 1
CODlist<- read_excel("CodQ1.xlsx")

# Right-Join Bill hoa don COD1: https://medium.com/@HollyEmblem/joining-data-with-dplyr-in-r-874698eb8898

RawD1 <- right_join(RawBill,CODlist,by="Cod")

# Filter Volume ; Success ; Delay and 

# Rename Colume -> English

RawD1<-RawD1 %>% rename(Volume="SL H", Success = "SL H  i sot", Delay="SL H delay", Returned="SL H  tr", TimeStamp="TG to", Money = "Tng tin thu") 

# filter theo CART
Extract_Cart <-RawD1 %>% 
  select(Cart, Cod,Volume, Success, Delay, Returned) %>% 
  group_by(Cart) %>%
  arrange(desc(Volume),desc(Success), Delay) %>% 
  summarise(
    TotalVolume=sum(Volume, na.rm = TRUE), 
    TotalSuccess=sum(Success,na.rm = TRUE), 
    TotalDelay=sum(Delay, na.rm = TRUE), 
    TotalReturned=sum(Returned,na.rm = TRUE),
    RatioSuccess = TotalSuccess/TotalVolume 
    )
  

Exportsheet<- list("Cart" = Extract_Cart, "Raw" = RawD1, "Cod" = CODlist) #assume sheet1 and sheet2 are data frames

write_xlsx(Exportsheet,"ExportSheet.xlsx")
